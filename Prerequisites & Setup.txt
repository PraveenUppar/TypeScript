
# For Quick testing
npm install -g typescript
tsc --version 
tsc --init
npm install -g ts-node

# For large project
npm install -g typescript
tsc --version 
tsc --init
npm init -y
npm install --save-dev typescript @types/node ts-node

# Commands
To run the ts file = "ts-node index.ts",
To build the ts file in dist: "tsc",
To run the js from the dist: "node dist/index.js"
